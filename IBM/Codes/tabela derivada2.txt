SELECT   DISTINCT 'RIVALDO' AS DIRETORIA, 
  DW.GDWKCORCDNGRLDIM.CDNGRLCOD, 
  DW.GDWKCORCDNGRLDIM.CDNGRLNOM, 
  DMCONTROLADORIA.CDMKSUCDIM.SUCCOD, 
  DMCONTROLADORIA.CDMKSUCDIM.SUCNOM 
FROM 
  DW.GDWKCORCDNGRLDIM,    -- * 
  DMCONTROLADORIA.CDMKSUCDIM,          -- * 
  DW.PDMKCOROGNDIM        -- * 
WHERE 
        ( DW.PDMKCOROGNDIM.SUCCOD=to_number(DMCONTROLADORIA.CDMKSUCDIM.SUCCOD)   AND dmcontroladoria.CDMKSUCDIM.SUCSTTCOD = 'A'  ) 
  AND   ( DW.GDWKCORCDNGRLDIM.CDNGRLCOD=DW.PDMKCOROGNDIM.CDNGRLCOD AND GDWKCORCDNGRLDIM.REGFNLDAT = TO_DATE('31/12/2099','DD/MM/YYYY')  ) 
  AND   ( DW.GDWKCORCDNGRLDIM.CDNGRLCOD  In  ( 10, 19, 20, 13, 47  )  AND  DW.PDMKCOROGNDIM.ANOMESREF  =  201109 ) 


union all
